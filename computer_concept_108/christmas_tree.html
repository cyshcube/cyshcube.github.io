    int n;
    cout << "Please enter n:";
    while(cin>>n)
    {
        if(n==0||n==1)
        {
            cout << "The Christmas tree can't hold the weight, try another number !" << endl << "Please enter n:";
            continue;
        }
        cout << "-----------------" << endl;
        for(int i=1;i<=n;i++)
        {
            for(int j=i;j<=i+2;j++)
            {
                for(int k=1;k<=n-j+2;k++)
                    cout << " ";
                for(int k=1;k<=2*j-1;k++)
                    cout << "*";
                cout << endl;
            }
        }
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=2+n/2;j++)
                cout << " ";
            for(int j=1;j<=(n%2==0?n-1:n);j++)
                cout << "*";
            cout << endl;
        }
        cout << "Please enter n:";
    }
